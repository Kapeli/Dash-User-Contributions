112 /*
113  * Helper functions to compute gradients-dot-residualvectors (1D to 4D)
114  * Note that these generate gradients of more than unit length. To make
115  * a close match with the value range of classic Perlin noise, the final
116  * noise values need to be rescaled to fit nicely within [-1,1].
117  * (The simplex noise functions as such also have different scaling.)
118  * Note also that these noise functions are the most practical and useful
119  * signed version of Perlin noise. To return values according to the
120  * RenderMan specification from the SL noise() and pnoise() functions,
121  * the noise values need to be scaled and offset to [0,1], like this:
122  * float SLnoise = (SimplexNoise1234::noise(x,y,z) + 1.0) * 0.5;
123  */
124 
125 inline float grad1( int hash, float x ) {
126  int h = hash & 15;
127  float grad = 1.0f + (h & 7); /* Gradient value 1.0, 2.0, ..., 8.0 */
128  if (h&8) grad = -grad; /* Set a random sign for the gradient */
129  return ( grad * x ); /* Multiply the gradient with the distance */
130 }
131 
132 inline float grad2( int hash, float x, float y ) {
133  int h = hash & 7; /* Convert low 3 bits of hash code */
134  float u = h<4 ? x : y; /* into 8 simple gradient directions, */
135  float v = h<4 ? y : x; /* and compute the dot product with (x,y). */
136  return ((h&1)? -u : u) + ((h&2)? -2.0f*v : 2.0f*v);
137 }
138 
139 inline float grad3( int hash, float x, float y , float z ) {
140  int h = hash & 15; /* Convert low 4 bits of hash code into 12 simple */
141  float u = h<8 ? x : y; /* gradient directions, and compute dot product. */
142  float v = h<4 ? y : h==12||h==14 ? x : z; /* Fix repeats at h = 12 to 15 */
143  return ((h&1)? -u : u) + ((h&2)? -v : v);
144 }
145 
146 inline float grad4( int hash, float x, float y, float z, float t ) {
147  int h = hash & 31; /* Convert low 5 bits of hash code into 32 simple */
148  float u = h<24 ? x : y; /* gradient directions, and compute dot product. */
149  float v = h<16 ? y : z;
150  float w = h<8 ? z : t;
151  return ((h&1)? -u : u) + ((h&2)? -v : v) + ((h&4)? -w : w);
152 }

168 /* 1D simplex noise */
169 inline float _slang_library_noise1 (float x)
170 {
171  int i0 = OFNOISE_FASTFLOOR(x);
172  int i1 = i0 + 1;
173  float x0 = x - i0;
174  float x1 = x0 - 1.0f;
175  float t1 = 1.0f - x1*x1;
176  float n0, n1;
177 
178  float t0 = 1.0f - x0*x0;
179 /* if(t0 < 0.0f) t0 = 0.0f; // this never happens for the 1D case */
180  t0 *= t0;
181  n0 = t0 * t0 * grad1(perm[i0 & 0xff], x0);
182 
183 /* if(t1 < 0.0f) t1 = 0.0f; // this never happens for the 1D case */
184  t1 *= t1;
185  n1 = t1 * t1 * grad1(perm[i1 & 0xff], x1);
186  /* The maximum value of this noise is 8*(3/4)^4 = 2.53125 */
187  /* A factor of 0.395 would scale to fit exactly within [-1,1], but */
188  /* we want to match PRMan's 1D noise, so we scale it down some more. */
189  return 0.25f * (n0 + n1);
190 }

192 /* 2D simplex noise */
193 inline float _slang_library_noise2 (float x, float y)
194 {
195  constexpr float F2 = 0.366025403f; /* F2 = 0.5*(sqrt(3.0)-1.0) */
196  constexpr float G2 = 0.211324865f; /* G2 = (3.0-Math.sqrt(3.0))/6.0 */
197 
198  float n0, n1, n2; /* Noise contributions from the three corners */
199 
200  /* Skew the input space to determine which simplex cell we're in */
201  float s = (x+y)*F2; /* Hairy factor for 2D */
202  float xs = x + s;
203  float ys = y + s;
204  int i = OFNOISE_FASTFLOOR(xs);
205  int j = OFNOISE_FASTFLOOR(ys);
206 
207  float t = (float)(i+j)*G2;
208  float X0 = i-t; /* Unskew the cell origin back to (x,y) space */
209  float Y0 = j-t;
210  float x0 = x-X0; /* The x,y distances from the cell origin */
211  float y0 = y-Y0;
212 
213  float x1, y1, x2, y2;
214  int ii, jj;
215  float t0, t1, t2;
216 
217  /* For the 2D case, the simplex shape is an equilateral triangle. */
218  /* Determine which simplex we are in. */
219  int i1, j1; /* Offsets for second (middle) corner of simplex in (i,j) coords */
220  if(x0>y0) {i1=1; j1=0;} /* lower triangle, XY order: (0,0)->(1,0)->(1,1) */
221  else {i1=0; j1=1;} /* upper triangle, YX order: (0,0)->(0,1)->(1,1) */
222 
223  /* A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and */
224  /* a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where */
225  /* c = (3-sqrt(3))/6 */
226 
227  x1 = x0 - i1 + G2; /* Offsets for middle corner in (x,y) unskewed coords */
228  y1 = y0 - j1 + G2;
229  x2 = x0 - 1.0f + 2.0f * G2; /* Offsets for last corner in (x,y) unskewed coords */
230  y2 = y0 - 1.0f + 2.0f * G2;
231 
232  /* Wrap the integer indices at 256, to avoid indexing perm[] out of bounds */
233  ii = i % 256;
234  jj = j % 256;
235 
236  /* Calculate the contribution from the three corners */
237  t0 = 0.5f - x0*x0-y0*y0;
238  if(t0 < 0.0f) n0 = 0.0f;
239  else {
240  t0 *= t0;
241  n0 = t0 * t0 * grad2(perm[ii+perm[jj]], x0, y0);
242  }
243 
244  t1 = 0.5f - x1*x1-y1*y1;
245  if(t1 < 0.0f) n1 = 0.0f;
246  else {
247  t1 *= t1;
248  n1 = t1 * t1 * grad2(perm[ii+i1+perm[jj+j1]], x1, y1);
249  }
250 
251  t2 = 0.5f - x2*x2-y2*y2;
252  if(t2 < 0.0f) n2 = 0.0f;
253  else {
254  t2 *= t2;
255  n2 = t2 * t2 * grad2(perm[ii+1+perm[jj+1]], x2, y2);
256  }
257 
258  /* Add contributions from each corner to get the final noise value. */
259  /* The result is scaled to return values in the interval [-1,1]. */
260  return 40.0f * (n0 + n1 + n2); /* TODO: The scale factor is preliminary! */
261 }

264 inline float _slang_library_noise3 (float x, float y, float z)
265 {
266  /* Simple skewing factors for the 3D case */
267  constexpr float F3 = 0.333333333f;
268  constexpr float G3 = 0.166666667f;
269 
270  float n0, n1, n2, n3; /* Noise contributions from the four corners */
271 
272  /* Skew the input space to determine which simplex cell we're in */
273  float s = (x+y+z)*F3; /* Very nice and simple skew factor for 3D */
274  float xs = x+s;
275  float ys = y+s;
276  float zs = z+s;
277  int i = OFNOISE_FASTFLOOR(xs);
278  int j = OFNOISE_FASTFLOOR(ys);
279  int k = OFNOISE_FASTFLOOR(zs);
280 
281  float t = (float)(i+j+k)*G3;
282  float X0 = i-t; /* Unskew the cell origin back to (x,y,z) space */
283  float Y0 = j-t;
284  float Z0 = k-t;
285  float x0 = x-X0; /* The x,y,z distances from the cell origin */
286  float y0 = y-Y0;
287  float z0 = z-Z0;
288 
289  float x1, y1, z1, x2, y2, z2, x3, y3, z3;
290  int ii, jj, kk;
291  float t0, t1, t2, t3;
292 
293  /* For the 3D case, the simplex shape is a slightly irregular tetrahedron. */
294  /* Determine which simplex we are in. */
295  int i1, j1, k1; /* Offsets for second corner of simplex in (i,j,k) coords */
296  int i2, j2, k2; /* Offsets for third corner of simplex in (i,j,k) coords */
297 
298 /* This code would benefit from a backport from the GLSL version! */
299  if(x0>=y0) {
300  if(y0>=z0)
301  { i1=1; j1=0; k1=0; i2=1; j2=1; k2=0; } /* X Y Z order */
302  else if(x0>=z0) { i1=1; j1=0; k1=0; i2=1; j2=0; k2=1; } /* X Z Y order */
303  else { i1=0; j1=0; k1=1; i2=1; j2=0; k2=1; } /* Z X Y order */
304  }
305  else { /* x0<y0 */
306  if(y0<z0) { i1=0; j1=0; k1=1; i2=0; j2=1; k2=1; } /* Z Y X order */
307  else if(x0<z0) { i1=0; j1=1; k1=0; i2=0; j2=1; k2=1; } /* Y Z X order */
308  else { i1=0; j1=1; k1=0; i2=1; j2=1; k2=0; } /* Y X Z order */
309  }
310 
311  /* A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z), */
312  /* a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and */
313  /* a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where */
314  /* c = 1/6. */
315 
316  x1 = x0 - i1 + G3; /* Offsets for second corner in (x,y,z) coords */
317  y1 = y0 - j1 + G3;
318  z1 = z0 - k1 + G3;
319  x2 = x0 - i2 + 2.0f*G3; /* Offsets for third corner in (x,y,z) coords */
320  y2 = y0 - j2 + 2.0f*G3;
321  z2 = z0 - k2 + 2.0f*G3;
322  x3 = x0 - 1.0f + 3.0f*G3; /* Offsets for last corner in (x,y,z) coords */
323  y3 = y0 - 1.0f + 3.0f*G3;
324  z3 = z0 - 1.0f + 3.0f*G3;
325 
326  /* Wrap the integer indices at 256, to avoid indexing perm[] out of bounds */
327  ii = i % 256;
328  jj = j % 256;
329  kk = k % 256;
330 
331  /* Calculate the contribution from the four corners */
332  t0 = 0.6f - x0*x0 - y0*y0 - z0*z0;
333  if(t0 < 0.0f) n0 = 0.0f;
334  else {
335  t0 *= t0;
336  n0 = t0 * t0 * grad3(perm[ii+perm[jj+perm[kk]]], x0, y0, z0);
337  }
338 
339  t1 = 0.6f - x1*x1 - y1*y1 - z1*z1;
340  if(t1 < 0.0f) n1 = 0.0f;
341  else {
342  t1 *= t1;
343  n1 = t1 * t1 * grad3(perm[ii+i1+perm[jj+j1+perm[kk+k1]]], x1, y1, z1);
344  }
345 
346  t2 = 0.6f - x2*x2 - y2*y2 - z2*z2;
347  if(t2 < 0.0f) n2 = 0.0f;
348  else {
349  t2 *= t2;
350  n2 = t2 * t2 * grad3(perm[ii+i2+perm[jj+j2+perm[kk+k2]]], x2, y2, z2);
351  }
352 
353  t3 = 0.6f - x3*x3 - y3*y3 - z3*z3;
354  if(t3<0.0f) n3 = 0.0f;
355  else {
356  t3 *= t3;
357  n3 = t3 * t3 * grad3(perm[ii+1+perm[jj+1+perm[kk+1]]], x3, y3, z3);
358  }
359 
360  /* Add contributions from each corner to get the final noise value. */
361  /* The result is scaled to stay just inside [-1,1] */
362  return 32.0f * (n0 + n1 + n2 + n3); /* TODO: The scale factor is preliminary! */
363 }

365 /* 4D simplex noise */
366 inline float _slang_library_noise4 (float x, float y, float z, float w)
367 {
368  /* The skewing and unskewing factors are hairy again for the 4D case */
369  constexpr float F4 = 0.309016994f; /* F4 = (Math.sqrt(5.0)-1.0)/4.0 */
370  constexpr float G4 = 0.138196601f; /* G4 = (5.0-Math.sqrt(5.0))/20.0 */
371 
372  float n0, n1, n2, n3, n4; /* Noise contributions from the five corners */
373 
374  /* Skew the (x,y,z,w) space to determine which cell of 24 simplices we're in */
375  float s = (x + y + z + w) * F4; /* Factor for 4D skewing */
376  float xs = x + s;
377  float ys = y + s;
378  float zs = z + s;
379  float ws = w + s;
380  int i = OFNOISE_FASTFLOOR(xs);
381  int j = OFNOISE_FASTFLOOR(ys);
382  int k = OFNOISE_FASTFLOOR(zs);
383  int l = OFNOISE_FASTFLOOR(ws);
384 
385  float t = (i + j + k + l) * G4; /* Factor for 4D unskewing */
386  float X0 = i - t; /* Unskew the cell origin back to (x,y,z,w) space */
387  float Y0 = j - t;
388  float Z0 = k - t;
389  float W0 = l - t;
390 
391  float x0 = x - X0; /* The x,y,z,w distances from the cell origin */
392  float y0 = y - Y0;
393  float z0 = z - Z0;
394  float w0 = w - W0;
395 
396  /* For the 4D case, the simplex is a 4D shape I won't even try to describe. */
397  /* To find out which of the 24 possible simplices we're in, we need to */
398  /* determine the magnitude ordering of x0, y0, z0 and w0. */
399  /* The method below is a good way of finding the ordering of x,y,z,w and */
400  /* then find the correct traversal order for the simplex we're in. */
401  /* First, six pair-wise comparisons are performed between each possible pair */
402  /* of the four coordinates, and the results are used to add up binary bits */
403  /* for an integer index. */
404  int c1 = (x0 > y0) ? 32 : 0;
405  int c2 = (x0 > z0) ? 16 : 0;
406  int c3 = (y0 > z0) ? 8 : 0;
407  int c4 = (x0 > w0) ? 4 : 0;
408  int c5 = (y0 > w0) ? 2 : 0;
409  int c6 = (z0 > w0) ? 1 : 0;
410  int c = c1 + c2 + c3 + c4 + c5 + c6;
411 
412  int i1, j1, k1, l1; /* The integer offsets for the second simplex corner */
413  int i2, j2, k2, l2; /* The integer offsets for the third simplex corner */
414  int i3, j3, k3, l3; /* The integer offsets for the fourth simplex corner */
415 
416  float x1, y1, z1, w1, x2, y2, z2, w2, x3, y3, z3, w3, x4, y4, z4, w4;
417  int ii, jj, kk, ll;
418  float t0, t1, t2, t3, t4;

420  /* simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some order. */
421  /* Many values of c will never occur, since e.g. x>y>z>w makes x<z, y<w and x<w */
422  /* impossible. Only the 24 indices which have non-zero entries make any sense. */
423  /* We use a thresholding to set the coordinates in turn from the largest magnitude. */
424  /* The number 3 in the "simplex" array is at the position of the largest coordinate. */
425  i1 = simplex[c][0]>=3 ? 1 : 0;
426  j1 = simplex[c][1]>=3 ? 1 : 0;
427  k1 = simplex[c][2]>=3 ? 1 : 0;
428  l1 = simplex[c][3]>=3 ? 1 : 0;
429  /* The number 2 in the "simplex" array is at the second largest coordinate. */
430  i2 = simplex[c][0]>=2 ? 1 : 0;
431  j2 = simplex[c][1]>=2 ? 1 : 0;
432  k2 = simplex[c][2]>=2 ? 1 : 0;
433  l2 = simplex[c][3]>=2 ? 1 : 0;
434  /* The number 1 in the "simplex" array is at the second smallest coordinate. */
435  i3 = simplex[c][0]>=1 ? 1 : 0;
436  j3 = simplex[c][1]>=1 ? 1 : 0;
437  k3 = simplex[c][2]>=1 ? 1 : 0;
438  l3 = simplex[c][3]>=1 ? 1 : 0;
439  /* The fifth corner has all coordinate offsets = 1, so no need to look that up. */
440 
441  x1 = x0 - i1 + G4; /* Offsets for second corner in (x,y,z,w) coords */
442  y1 = y0 - j1 + G4;
443  z1 = z0 - k1 + G4;
444  w1 = w0 - l1 + G4;
445  x2 = x0 - i2 + 2.0f*G4; /* Offsets for third corner in (x,y,z,w) coords */
446  y2 = y0 - j2 + 2.0f*G4;
447  z2 = z0 - k2 + 2.0f*G4;
448  w2 = w0 - l2 + 2.0f*G4;
449  x3 = x0 - i3 + 3.0f*G4; /* Offsets for fourth corner in (x,y,z,w) coords */
450  y3 = y0 - j3 + 3.0f*G4;
451  z3 = z0 - k3 + 3.0f*G4;
452  w3 = w0 - l3 + 3.0f*G4;
453  x4 = x0 - 1.0f + 4.0f*G4; /* Offsets for last corner in (x,y,z,w) coords */
454  y4 = y0 - 1.0f + 4.0f*G4;
455  z4 = z0 - 1.0f + 4.0f*G4;
456  w4 = w0 - 1.0f + 4.0f*G4;
457 
458  /* Wrap the integer indices at 256, to avoid indexing perm[] out of bounds */
459  ii = i % 256;
460  jj = j % 256;
461  kk = k % 256;
462  ll = l % 256;
463 
464  /* Calculate the contribution from the five corners */
465  t0 = 0.6f - x0*x0 - y0*y0 - z0*z0 - w0*w0;
466  if(t0 < 0.0f) n0 = 0.0f;
467  else {
468  t0 *= t0;
469  n0 = t0 * t0 * grad4(perm[ii+perm[jj+perm[kk+perm[ll]]]], x0, y0, z0, w0);
470  }
471 
472  t1 = 0.6f - x1*x1 - y1*y1 - z1*z1 - w1*w1;
473  if(t1 < 0.0f) n1 = 0.0f;
474  else {
475  t1 *= t1;
476  n1 = t1 * t1 * grad4(perm[ii+i1+perm[jj+j1+perm[kk+k1+perm[ll+l1]]]], x1, y1, z1, w1);
477  }
478 
479  t2 = 0.6f - x2*x2 - y2*y2 - z2*z2 - w2*w2;
480  if(t2 < 0.0f) n2 = 0.0f;
481  else {
482  t2 *= t2;
483  n2 = t2 * t2 * grad4(perm[ii+i2+perm[jj+j2+perm[kk+k2+perm[ll+l2]]]], x2, y2, z2, w2);
484  }
485 
486  t3 = 0.6f - x3*x3 - y3*y3 - z3*z3 - w3*w3;
487  if(t3 < 0.0f) n3 = 0.0f;
488  else {
489  t3 *= t3;
490  n3 = t3 * t3 * grad4(perm[ii+i3+perm[jj+j3+perm[kk+k3+perm[ll+l3]]]], x3, y3, z3, w3);
491  }
492 
493  t4 = 0.6f - x4*x4 - y4*y4 - z4*z4 - w4*w4;
494  if(t4 < 0.0f) n4 = 0.0f;
495  else {
496  t4 *= t4;
497  n4 = t4 * t4 * grad4(perm[ii+1+perm[jj+1+perm[kk+1+perm[ll+1]]]], x4, y4, z4, w4);
498  }
499 
500  /* Sum up and scale the result to cover the range [-1,1] */
501  return 27.0f * (n0 + n1 + n2 + n3 + n4); /* TODO: The scale factor is preliminary! */
