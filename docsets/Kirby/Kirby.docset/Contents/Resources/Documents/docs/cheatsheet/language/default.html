$language->default()

Checks if this is the default language

type: boolean
text:

Example

<ul>
  <?php foreach($site->languages() as $language): ?>
  <li>
    <a href="<?php echo $language->url() ?>">
      <?php echo $language->code() ?>
      <?php e($language->default(), ' (default language)') ?>
    </a>
  </li>
  <?php endforeach ?>
</ul>
