$page->isActive()

Checks if the page is the currently active page.

type: boolean
text:

Example

<nav role="navigation">
  <ul>
    <?php foreach($pages as $subpage): ?>
    <li<?php e($subpage->isActive(), ' class="active"') ?>>
      <a href="<?php echo $subpage->url() ?>">
        <?php echo html($subpage->title()) ?>
      </a>
    </li>
    <?php endforeach ?>
  </ul>
</nav>
